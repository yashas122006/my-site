#include <stdio.h>
#include <stdlib.h>
#define M 2
char q[M]; int f = -1, r = -1, ch;
void insert() {
if((r+1)%M == f) printf("Overflow\n");
else {
char c; printf("Enter the element to be inserted:\n"); 
scanf(" %c", &c);
if(f == -1) f = 0;
r = (r + 1) % M; q[r] = c;
}
}
void delete() {
if(f == -1) printf("Underflow\n");
else {
printf("Deleted element: %c\n", q[f]); 
if(f==r) f= r=-1;
else f = (f+1) % M;
}
}
void display() {
if(f == -1) printf("Queue is Empty\n");
else {
int i = f; while(i!= r) { printf("%c", q[i]); i = (i+1) % M; } printf("%c\n", q[r]);
}
}
int main() {
while(1) {
printf("1.Insert 2. Delete 3. Display 4.Exit\n\n Enter Your Choice:\n");
scanf("%d",&ch); 
if(ch==1) insert();
else if(ch==2) delete(); 
else if(ch==3) display(); 
else if(ch==4) exit(0);
else printf("invalid choice");
}
return 0;
}

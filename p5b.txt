#include <stdio.h>
void hanoi(int n, char s, char t, char d){
if (n == 0) return;
hanoi(n - 1, s, d, t);
printf(" MOVE DISK %d FROM PEG %c TO PEG %c\n", n, s, d);
hanoi(n-1, t, s, d);
}
int main() {
int n;
printf("Run 1:\nenter the number of disks: ");
scanf("%d", &n);
printf("The sequence of moves involved in the Tower of Hanoi are:\n\n");
hanoi(n, 'A', 'B', 'C');
return 0;
}
